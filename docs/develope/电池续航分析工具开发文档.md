class BatteryAnalyzer {
    
    // è®¡ç®—ç»­èˆªæ—¶é—´
    fun calculateBatteryLife(records: List<BatteryRecord>): Duration {
        if (records.size < 2) return Duration.ZERO
        val startTime = records.first().timestamp
        val endTime = records.last().timestamp
        return Duration.ofMillis(endTime - startTime)
    }
    
    // è®¡ç®—æ”¾ç”µé€Ÿç‡ (%/å°æ—¶)
    fun calculateDischargeRate(records: List<BatteryRecord>): Float {
        if (records.size < 2) return 0f
        val levelDrop = records.first().level - records.last().level
        val hours = calculateBatteryLife(records).toHours().toFloat()
        return if (hours > 0) levelDrop / hours else 0f
    }
    
    // é¢„æµ‹å‰©ä½™ç»­èˆªæ—¶é—´
    fun predictRemainingTime(
        currentLevel: Int,
        recentRecords: List<BatteryRecord>
    ): Duration {
        val rate = calculateDischargeRate(recentRecords)
        if (rate <= 0) return Duration.ofHours(99)
        val hoursRemaining = currentLevel / rate
        return Duration.ofMinutes((hoursRemaining * 60).toLong())
    }
}
